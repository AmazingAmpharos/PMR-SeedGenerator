This file describes the setup of the SQLite database used to provide default values to the randomizer.
The information below is written is pseudo-code to roughly describe the data within each table
and column.

# each row is map, eg. kmr.kmr_02
DESC TABLE MapArea (
  area_id       INT              # eg 0 (=kmr)
  map_id        INT              # eg 1 (=kmr_02)
  name          VARCHAR NULLABLE # eg 'kmr_02'
  verbose_name  VARCHAR NULLABLE # descriptive location name
)

# each row is an actor attribute
# in this case each attribute is either level, hp, or damagevalues of enemies
DESC TABLE ActorAttribute (
  actor_name  VARCHAR    # human readable name
  attribute   VARCHAR    # hp, level, damage etc.
  value       INT        # value of attribute
  map_area    FK -> MapArea
)

# each row describes general info about an item
DESC TABLE Item (
  item_type          VARCHAR NULLABLE          # keyitem, badge etc
  value              INT                       # item byte value as int (eg 348 = 0x15c (StarPiece))
  item_name          VARCHAR                   # actual item name w/o spaces or apostrophe
  base_price         INT                       # default cost of item if location is a shop
  progression        BOOL                      # True if item can be required to reach locations
)

# each row describes a node in the world graph
DESC TABLE Node (
  map_area        FK -> MapArea
  entrance_id     INT NULLABLE          # entrance_id of map entrance
  entrance_type   VARCHAR NULLABLE      # door, walk, pipe etc.
  entrance_name   VARCHAR NULLABLE      # verbose description of entrance
  key_name_item   VARCHAR NULLABLE      # human readable name of item location (eg ItemA)
  key_name_price  VARCHAR NULLABLE      # human readable name of item price in shop (eg ShopPriceA)
  vanilla_item    FK -> Item NULLABLE   # reference to item placed here in the unmodified game
  current_item    FK -> Item NULLABLE   # reference to item placed here during randomization
  item_index      INT NULLABLE          # 
  price_index     INT NULLABLE          # 
)

# each row describes an edge in the world graph
DESC TABLE Edge (
  name
  origin_node   FK -> Node  # the node this edge leads away from
  target_node   FK -> Node  # the node this edge leads towards
  requirements  JSON        # requirements to traverse this edge
)

# each table is an option for the randomizer
DESC TABLE Option (
  name    VARCHAR  #
  value   INT      #
)

# each row is a question asked by chuck quizmo?
DESC TABLE Quiz (
  name      VARCHAR       #
  value     INT           #
  map_area  FK -> MapArea # location of Chuck Quizmo
)
